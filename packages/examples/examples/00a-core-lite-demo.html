The Code

This entire demo uses just basic pub/sub - no framework needed!
// Import core-lite (9KB)
import '@larcjs/core-lite';

// Subscribe to counter state
document.addEventListener('pan:deliver', (e) => {
  if (e.detail.topic === 'counter.state') {
    document.getElementById('counter').textContent = e.detail.data;
  }
});

// Publish increment
document.getElementById('incrementBtn').addEventListener('click', () => {
  document.dispatchEvent(new CustomEvent('pan:publish', {
    detail: { topic: 'counter.increment', data: 1 }
  }));
});
